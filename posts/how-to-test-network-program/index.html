    #include <pcap.h>
#include <stdio.h>
#include <stdio_ext.h>
#include <string.h>
#include <unistd.h>

#define MAX_NAME_LEN 256
#define EXIT_IF_TRUE(STATE, ARGS...) ({ \
  if(STATE) { \
    fprintf(stderr, ARGS); \
    return 1; \
  } \
})

pcap_t *ppcap = NULL;
pcap_dumper_t *pdump = NULL;

void cleanup() {
  pcap_breakloop(ppcap);
  pcap_close(ppcap);
}

int main(int argc, char *argv[]) {
  char interface_name[MAX_NAME_LEN];
  char output_filename[MAX_NAME_LEN];
  char errbuf[PCAP_ERRBUF_SIZE];
  struct bpf_program prog;
  pid_t pid = 0;

  EXIT_IF_TRUE(argc != 3, "[!] Usage : %s [Interface] [Output Filename]\n", argv[0]);

  strcpy(interface_name, argv[1]);
  strcpy(output_filename, argv[2]);

  EXIT_IF_TRUE((ppcap = pcap_open_live(interface_name, BUFSIZ, 1, 1000, errbuf)) == NULL,
    "[!] Failed to open interface %s : %s\n", interface_name, errbuf);
  EXIT_IF_TRUE((pdump = pcap_dump_open(ppcap, output_filename)) == NULL,
    "[!] Failed to create %s\n", output_filename);
  EXIT_IF_TRUE(pcap_compile(ppcap, &prog, "tcp or udp or icmp", 0, 0) < 0,
    "[!] Error compiling bpf filter on %s : %s\n", interface_name, pcap_geterr(ppcap));
  EXIT_IF_TRUE(pcap_setfilter(ppcap, &prog) < 0,
    "[!] Error installing bpf filter on interface %s\n", interface_name);

  printf("------------------------------\n");
  printf("Target Interface : %s\n", interface_name);
  printf("Output File : %s\n", output_filename);
  printf("------------------------------\n");
  printf("[i] Simple packet capture start\n");
  printf("[i] Insert 'q' to quit capture\n");

  pid = fork();
  EXIT_IF_TRUE(pid == -1, "[!] Failed to create child process");
  if(pid == 0) {
    pcap_loop(ppcap, -1, &pcap_dump, (char*)pdump);

    return 0;
  }

  do {
    __fpurge(stdin);
    printf("cmd : ");
  } while(getchar() != 'q');
  cleanup();
  return 0;
}
