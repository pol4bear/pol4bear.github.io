네트워크 프로그램 테스트하기

패킷 로스 등 네트워크 프로그램이 정상 작동하는지 확인하기

Abstract#

내가 짠 네트워크 프로그램에 혹시 패킷 로스가 있지는 않을까? 가상 인터페이스와 TCPReplay를 이용해서 확인해보도록 하자

01. 가상 인터페이스 추가#

  • 가상 인터페이스 또는, 더미 인터페이스는 실제 설치되어있지 않지만 마치 네트워크 인터페이스와 같은 역할을 한다.

  • 일반 네트워크 인터페이스를 사용할 수 도 있지만, 이미 사용되고 있는 인터페이스의 경우 불필요한 트래픽들이 많이 잡히기 때문에 가상 인터페이스를 사용한다.

  • 아래 명령어로 가상 인터페이스를 추가할 수 있다.

    # 인터페이스 추가
sudo ip link add dum0 type dummy
# 인터페이스 활성화
sudo ifconfig dum0 up

02 . 테스트 프로그램 준비#

  • 패킷 로스가 있는지 확인할 프로그램(이하 ‘테스트 프로그램’이라고 한다.)과 패킷 파일을 준비한다.

  • 패킷은 웹 사이트 접속 패킷을 잡은 이 파일을 사용한다.

  • 테스트 프로그램은 인터페이스에서 잡힌 패킷을 단순 저장하는 이 프로그램을 사용한다.

    #include <pcap.h>
#include <stdio.h>
#include <stdio_ext.h>
#include <string.h>
#include <unistd.h>

#define MAX_NAME_LEN 256
#define EXIT_IF_TRUE(STATE, ARGS...) ({ \
  if(STATE) { \
    fprintf(stderr, ARGS); \
    return 1; \
  } \
})

pcap_t *ppcap = NULL;
pcap_dumper_t *pdump = NULL;

void cleanup() {
  pcap_breakloop(ppcap);
  pcap_close(ppcap);
}

int main(int argc, char *argv[]) {
  char interface_name[MAX_NAME_LEN];
  char output_filename[MAX_NAME_LEN];
  char errbuf[PCAP_ERRBUF_SIZE];
  struct bpf_program prog;
  pid_t pid = 0;

  EXIT_IF_TRUE(argc != 3, "[!] Usage : %s [Interface] [Output Filename]\n", argv[0]);

  strcpy(interface_name, argv[1]);
  strcpy(output_filename, argv[2]);

  EXIT_IF_TRUE((ppcap = pcap_open_live(interface_name, BUFSIZ, 1, 1000, errbuf)) == NULL,
    "[!] Failed to open interface %s : %s\n", interface_name, errbuf);
  EXIT_IF_TRUE((pdump = pcap_dump_open(ppcap, output_filename)) == NULL,
    "[!] Failed to create %s\n", output_filename);
  EXIT_IF_TRUE(pcap_compile(ppcap, &prog, "tcp or udp or icmp", 0, 0) < 0,
    "[!] Error compiling bpf filter on %s : %s\n", interface_name, pcap_geterr(ppcap));
  EXIT_IF_TRUE(pcap_setfilter(ppcap, &prog) < 0,
    "[!] Error installing bpf filter on interface %s\n", interface_name);

  printf("------------------------------\n");
  printf("Target Interface : %s\n", interface_name);
  printf("Output File : %s\n", output_filename);
  printf("------------------------------\n");
  printf("[i] Simple packet capture start\n");
  printf("[i] Insert 'q' to quit capture\n");

  pid = fork();
  EXIT_IF_TRUE(pid == -1, "[!] Failed to create child process");
  if(pid == 0) {
    pcap_loop(ppcap, -1, &pcap_dump, (char*)pdump);

    return 0;
  }

  do {
    __fpurge(stdin);
    printf("cmd : ");
  } while(getchar() != 'q');
  cleanup();
  return 0;
}

03. 프로그램 테스트#

  • 테스트를 위해 프로그램을 실행한 후 TCPReplay로 패킷을 재전송한다.

    # -t, --topspeed : 전송할 수 있는 최대 속도로 전송
# -K : 캡처 파일을 메모리에 우선 로드한 후 전송
sudo tcpreplay -i dum0 -t -K test.pcapng

  • 패킷을 재전송할 때 Error with PF_PACKET send() [277]: Message too long (errno=90)같은 에러가 발생할 수 있는데

  • MTU를 늘리거나 tcpreplay-edit으로 MTU 크기만큼 잘라서 전송하도록 하면 된다.

    # MTU 늘리기
sudo ifconfig dum0 mtu 10000

# tcpreplay-edit 활용
sudo tcpreplay-edit -i dum0 --mtu-trunc -t -K test.pcapng
